446  cmat super(const std::vector<cmat> &Ks)
447  {
448  // EXCEPTION CHECKS
449  if (!internal::_check_nonzero_size(Ks))
450  throw Exception("super", Exception::Type::ZERO_SIZE);
451  if (!internal::_check_nonzero_size(Ks[0]))
452  throw Exception("super", Exception::Type::ZERO_SIZE);
453  if (!internal::_check_square_mat(Ks[0]))
454  throw Exception("super", Exception::Type::MATRIX_NOT_SQUARE);
455  for (auto &&it : Ks)
456  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
457  throw Exception("super", Exception::Type::DIMS_NOT_EQUAL);
458  std::size_t D = static_cast<std::size_t>(Ks[0].rows());
459 
460  std::size_t midx_row[2] = {0, 0};
461  std::size_t midx_col[2] = {0, 0};
462  std::size_t dims[2];
463  dims[0] = dims[1] = D;
464 
465  cmat result(D * D, D * D);
466  cmat MN = cmat::Zero(D, D);
467  cmat BA = cmat::Zero(D, D);
468  cmat EMN = cmat::Zero(D, D);
469 
470  for (std::size_t m = 0; m < D; ++m)
471  {
472  midx_col[0] = m;
473  for (std::size_t n = 0; n < D; ++n)
474  {
475  midx_col[1] = n;
476  MN(m, n) = 1;
477  // compute E(|m><n|)
478  for (std::size_t i = 0; i < Ks.size(); ++i)
479  EMN += Ks[i] * MN * adjoint(Ks[i]);
480  MN(m, n) = 0;
481  for (std::size_t a = 0; a < D; ++a)
482  {
483  midx_row[0] = a;
484  for (std::size_t b = 0; b < D; ++b)
485  {
486  midx_row[1] = b;
487  BA(b, a) = 1;
488 
489  // compute result(ab,mn)=<a|E(|m><n)|b>
490 
491  result(internal::_multiidx2n(midx_row, 2, dims),
492  internal::_multiidx2n(midx_col, 2, dims)) = (EMN
493  * BA).trace();
494  BA(b, a) = 0;
495  }
496  }
497  EMN = cmat::Zero(D, D);
498  }
499  }
500  return result;

678  cmat choi(const std::vector<cmat> &Ks)
679  {
680  // EXCEPTION CHECKS
681  if (!internal::_check_nonzero_size(Ks))
682  throw Exception("choi", Exception::Type::ZERO_SIZE);
683  if (!internal::_check_nonzero_size(Ks[0]))
684  throw Exception("choi", Exception::Type::ZERO_SIZE);
685  if (!internal::_check_square_mat(Ks[0]))
686  throw Exception("choi", Exception::Type::MATRIX_NOT_SQUARE);
687  for (auto &&it : Ks)
688  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
689  throw Exception("choi", Exception::Type::DIMS_NOT_EQUAL);
690  std::size_t D = static_cast<std::size_t>(Ks[0].rows());
691 
692  // construct the D x D \sum |jj> vector
693  // (un-normalized maximally entangled state)
694  cmat MES = cmat::Zero(D * D, 1);
695  for (std::size_t a = 0; a < D; ++a)
696  MES(a * D + a) = 1;
697 
698  cmat Omega = static_cast<cmat>(MES * adjoint(MES));
699 
700  cmat result = cmat::Zero(D * D, D * D);
701 
702  for (std::size_t i = 0; i < Ks.size(); ++i)
703  {
704  result += kron(cmat::Identity(D, D), Ks[i]) * Omega
705  * adjoint(kron(cmat::Identity(D, D), Ks[i]));
706  }
707 
708  return result;
709  }

796  if (std::abs((cplx) ev(i)) > max)
