38  std::vector<double> x2contfrac(double x, std::size_t n, std::size_t cut = 1e5)
39  {
40  if (n == 0)
41  throw Exception("x2contfrac", Exception::Type::OUT_OF_RANGE);
42 
43  std::vector<double> result;
44 
45  double r = x;
46  double a = std::floor(r);
47  for (std::size_t i = 0; i < n; ++i)
48  {
49  result.push_back(a);
50  r = 1. / (r - a);
51  a = std::floor(r);

53  return result;
54  }
