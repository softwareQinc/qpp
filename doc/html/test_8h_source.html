52 template<typename Derived>
53 std::tuple<std::vector<idx>, double, cmat>
54 measure_seq(
55  const Eigen::MatrixBase<Derived>& A,
56  std::vector<idx> subsys,
57  std::vector<idx> dims)
58 {
59  dyn_mat<typename Derived::Scalar> cA = A;
60 
61  // EXCEPTION CHECKS
62 
63  // check zero-size
64  if (!internal::_check_nonzero_size(cA))
65  throw Exception("qpp::measure_seq()", Exception::Type::ZERO_SIZE);
66 
67  // check that dimension is valid
68  if (!internal::_check_dims(dims))
69  throw Exception("qpp::measure_seq()", Exception::Type::DIMS_INVALID);
70 
71  // check that dims match rho matrix
72  if (!internal::_check_dims_match_mat(dims, cA))
73  throw Exception("qpp::measure_seq()",
74  Exception::Type::DIMS_MISMATCH_MATRIX);
75 
76  // check subsys is valid w.r.t. dims
77  if (!internal::_check_subsys_match_dims(subsys, dims))
78  throw Exception("qpp::measure_seq()",
79  Exception::Type::SUBSYS_MISMATCH_DIMS);
80 
81  // END EXCEPTION CHECKS
82 
83  std::vector<idx> result;
84  double prob = 1;
85 
86  // sort subsys in decreasing order,
87  // the order of measurements does not matter
88  std::sort(std::begin(subsys), std::end(subsys), std::greater<idx>{});
89 
90  //************ density matrix or column vector ************//
91  if (internal::_check_square_mat(cA) || internal::_check_col_vector(cA))
92  {
93  while (subsys.size() > 0)
94  {
95  auto tmp = measure(cA, gt.Id(dims.at(subsys.at(0))),
96  {subsys.at(0)}, dims);
97  result.push_back(std::get<0>(tmp));
98  prob *= std::get<1>(tmp).at(std::get<0>(tmp));
99  cA = std::get<2>(tmp).at(std::get<0>(tmp));
100 
101  // remove the subsystem
102  dims.erase(std::next(dims.begin(), subsys.at(0)));
103  subsys.erase(subsys.begin());
104  }
105  }
106  else
107  throw Exception("qpp::measure_seq()",
108  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
109 
110  return std::make_tuple(result, prob, cA);
111 }

142
