343  virtual void execute(const QCircuit::iterator::value_type& elem) {
344  // EXCEPTION CHECKS
345 
346  // iterator must point to the same quantum circuit
347  if (elem.value_type_qc_ != qc_)
348  throw exception::InvalidIterator("qpp::QEngine::execute()");
349  // the rest of exceptions are caught by the iterator::operator*()
350  // END EXCEPTION CHECKS
351 
352  auto h_tbl = qc_->get_cmat_hash_tbl_();
353 
354  // gate step
355  if (elem.type_ == QCircuit::StepType::GATE) {
356  auto gates = qc_->get_gates_();
357 
358  idx q_ip =
359  std::distance(std::begin(qc_->get_gates_()), elem.gates_ip_);
360 
361  std::vector<idx> ctrl_rel_pos;
362  std::vector<idx> target_rel_pos =
363  get_relative_pos_(gates[q_ip].target_);
364 
365  switch (gates[q_ip].gate_type_) {
366  case QCircuit::GateType::NONE:
367  break;
368  case QCircuit::GateType::SINGLE:
369  case QCircuit::GateType::TWO:
370  case QCircuit::GateType::THREE:
371  case QCircuit::GateType::CUSTOM:
372  st_.psi_ = apply(st_.psi_, h_tbl[gates[q_ip].gate_hash_],
373  target_rel_pos, qc_->get_d());
374  break;
375  case QCircuit::GateType::FAN:
376  for (idx m = 0; m < gates[q_ip].target_.size(); ++m)
377  st_.psi_ =
378  apply(st_.psi_, h_tbl[gates[q_ip].gate_hash_],
379  {target_rel_pos[m]}, qc_->get_d());
380  break;
381  case QCircuit::GateType::SINGLE_CTRL_SINGLE_TARGET:
382  case QCircuit::GateType::SINGLE_CTRL_MULTIPLE_TARGET:
383  case QCircuit::GateType::MULTIPLE_CTRL_SINGLE_TARGET:
384  case QCircuit::GateType::MULTIPLE_CTRL_MULTIPLE_TARGET:
385  case QCircuit::GateType::CUSTOM_CTRL:
386  ctrl_rel_pos = get_relative_pos_(gates[q_ip].ctrl_);
387  st_.psi_ =
388  applyCTRL(st_.psi_, h_tbl[gates[q_ip].gate_hash_],
389  ctrl_rel_pos, target_rel_pos, qc_->get_d());
390  break;
391  case QCircuit::GateType::SINGLE_cCTRL_SINGLE_TARGET:
392  case QCircuit::GateType::SINGLE_cCTRL_MULTIPLE_TARGET:
393  case QCircuit::GateType::MULTIPLE_cCTRL_SINGLE_TARGET:
394  case QCircuit::GateType::MULTIPLE_cCTRL_MULTIPLE_TARGET:
395  case QCircuit::GateType::CUSTOM_cCTRL:
396  if (st_.dits_.empty()) {
397  st_.psi_ =
398  apply(st_.psi_, h_tbl[gates[q_ip].gate_hash_],
399  target_rel_pos, qc_->get_d());
400  } else {
401  bool should_apply = true;
402  idx first_dit;
403  // we have a shift
404  if (!gates[q_ip].shift_.empty()) {
405  first_dit = st_.dits_[(gates[q_ip].ctrl_)[0]] +
406  gates[q_ip].shift_[0];
407  for (idx m = 1; m < gates[q_ip].ctrl_.size(); ++m) {
408  if (st_.dits_[(gates[q_ip].ctrl_)[m]] +
409  gates[q_ip].shift_[m] !=
410  first_dit) {
411  should_apply = false;
412  break;
413  }
414  }
415  }
416  // no shift
417  else {
418  first_dit = st_.dits_[(gates[q_ip].ctrl_)[0]];
419  for (idx m = 1; m < gates[q_ip].ctrl_.size(); ++m) {
420  if (st_.dits_[(gates[q_ip].ctrl_)[m]] !=
421  first_dit) {
422  should_apply = false;
423  break;
424  }
425  }
426  }
427 
428  if (should_apply) {
429  st_.psi_ = apply(
430  st_.psi_,
431  powm(h_tbl[gates[q_ip].gate_hash_], first_dit),
432  target_rel_pos, qc_->get_d());
433  }
434  }
435  break;
436  } // end switch on gate type
437  } // end if gate step
438 
439  // measurement step
440  else if (elem.type_ == QCircuit::StepType::MEASUREMENT) {
441  auto measurements = qc_->get_measurements_();
442  idx m_ip = std::distance(std::begin(qc_->get_measurements_()),
443  elem.measurements_ip_);
444 
445  std::vector<idx> target_rel_pos =
446  get_relative_pos_(measurements[m_ip].target_);
447 
448  std::vector<idx> resZ;
449  double probZ;
450 
451  idx mres = 0;
452  std::vector<double> probs;
453  std::vector<cmat> states;
454 
455  switch (measurements[m_ip].measurement_type_) {
456  case QCircuit::MeasureType::NONE:
457  break;
458  case QCircuit::MeasureType::MEASURE_Z:
459  std::tie(resZ, probZ, st_.psi_) =
460  measure_seq(st_.psi_, target_rel_pos, qc_->get_d());
461  st_.dits_[measurements[m_ip].c_reg_] = resZ[0];
462  st_.probs_[measurements[m_ip].c_reg_] = probZ;
463  set_measured_(measurements[m_ip].target_[0]);
464  break;
465  case QCircuit::MeasureType::MEASURE_Z_MANY:
466  std::tie(resZ, probZ, st_.psi_) =
467  measure_seq(st_.psi_, target_rel_pos, qc_->get_d());
468  st_.dits_[measurements[m_ip].c_reg_] =
469  multiidx2n(resZ, std::vector<idx>(target_rel_pos.size(),
470  qc_->get_d()));
471  st_.probs_[measurements[m_ip].c_reg_] = probZ;
472  for (auto&& elem : measurements[m_ip].target_)
473  set_measured_(elem);
474  break;
475  case QCircuit::MeasureType::MEASURE_V:
476  std::tie(mres, probs, states) = measure(
477  st_.psi_, h_tbl[measurements[m_ip].mats_hash_[0]],
478  target_rel_pos, qc_->get_d());
479  st_.psi_ = states[mres];
480  st_.dits_[measurements[m_ip].c_reg_] = mres;
481  st_.probs_[measurements[m_ip].c_reg_] = probs[mres];
482  set_measured_(measurements[m_ip].target_[0]);
483  break;
484  case QCircuit::MeasureType::MEASURE_V_MANY:
485  std::tie(mres, probs, states) = measure(
486  st_.psi_, h_tbl[measurements[m_ip].mats_hash_[0]],
487  target_rel_pos, qc_->get_d());
488  st_.psi_ = states[mres];
489  st_.dits_[measurements[m_ip].c_reg_] = mres;
490  st_.probs_[measurements[m_ip].c_reg_] = probs[mres];
491  for (auto&& elem : measurements[m_ip].target_)
492  set_measured_(elem);
493  break;
494  } // end switch on measurement type
495  } // end else if measurement step
496 
497  // no-op
498  else if (elem.type_ == QCircuit::StepType::NOP) {
499  }
500 
501  // otherwise
502  else {
503  }
504  }

520  void execute(idx reps = 1, bool clear_stats = true) {
521  auto initial_engine_state = st_; // saves the engine entry state
522 
523  if (clear_stats)
524  reset_stats();
525 
526  for (idx i = 0; i < reps; ++i) {
527  // sets the state of the engine to the entry state
528  st_ = initial_engine_state;
529 
530  for (auto&& elem : *qc_)
531  execute(elem);
532

535  std::vector<idx> m_res = get_dits();
