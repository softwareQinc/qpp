45 // integer index to multi-index, use C-style array for speed

47 inline void _n2multiidx(idx n, idx numdims, const idx* dims, idx* result)
48 noexcept
49 {
50  // no error checks to improve speed
51  for (idx i = 0; i < numdims; ++i)

53  result[numdims - i - 1] = n % (dims[numdims - i - 1]);
54  n /= (dims[numdims - i - 1]);
55  }
56 }

58 // multi-index to integer index, use C-style array for speed,
59 // standard lexicographical order, e.g. 00->0, 01->1, 10->2, 11->3
60 inline idx _multiidx2n(const idx* midx, idx numdims, const idx* dims)
61 noexcept
62 {
63  // no error checks to improve speed
64 
65  // Static allocation for speed!
66  // double the size for matrices reshaped as vectors
67  idx part_prod[2 * maxn];
68 
69  idx result = 0;
70  part_prod[numdims - 1] = 1;
71  for (idx i = 1; i < numdims; ++i)
72  {
73  part_prod[numdims - i - 1] =
74  part_prod[numdims - i] * dims[numdims - i];
75  result += midx[numdims - i - 1] * part_prod[numdims - i - 1];
76  }
77 
78  return result + midx[numdims - 1];
79 }

265 // internal function for the variadic template function wrapper kron()
266 template<typename Derived1, typename Derived2>
267 dyn_mat<typename Derived1::Scalar> _kron2(const Eigen::MatrixBase<Derived1>& A,
268  const Eigen::MatrixBase<Derived2>& B)
269 {
270  const dyn_mat<typename Derived1::Scalar>& rA = A;
271  const dyn_mat<typename Derived2::Scalar>& rB = B;
272 
273  // EXCEPTION CHECKS
274 
275  // check types
276  if (!std::is_same<typename Derived1::Scalar,
277  typename Derived2::Scalar>::value)
278  throw Exception("qpp::kron()", Exception::Type::TYPE_MISMATCH);
279 
280  // check zero-size
281  if (!internal::_check_nonzero_size(rA))
282  throw Exception("qpp::kron()", Exception::Type::ZERO_SIZE);
283 
284  // check zero-size
285  if (!internal::_check_nonzero_size(rB))
286  throw Exception("qpp::kron()", Exception::Type::ZERO_SIZE);
287 
288  idx Acols = static_cast<idx>(rA.cols());
289  idx Arows = static_cast<idx>(rA.rows());
290  idx Bcols = static_cast<idx>(rB.cols());
291  idx Brows = static_cast<idx>(rB.rows());
292 
293  dyn_mat<typename Derived1::Scalar> result;
294  result.resize(Arows * Brows, Acols * Bcols);
295 
296 #pragma omp parallel for collapse(2)
297  for (idx j = 0; j < Acols; ++j)
298  for (idx i = 0; i < Arows; ++i)
299  result.block(i * Brows, j * Bcols, Brows, Bcols) = rA(i, j) * rB;
300 
301  return result;
302 
303 }
