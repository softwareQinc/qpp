43 template<typename Derived>
44 dyn_col_vect<typename Derived::Scalar> ip(
45  const Eigen::MatrixBase<Derived>& phi,
46  const Eigen::MatrixBase<Derived>& psi,
47  const std::vector<idx>& subsys,
48  const std::vector<idx>& dims)
49 {
50  const dyn_col_vect<typename Derived::Scalar>& rphi = phi;
51  const dyn_col_vect<typename Derived::Scalar>& rpsi = psi;
52 
53  // EXCEPTION CHECKS
54 
55  // check zero-size
56  if (!internal::_check_nonzero_size(rphi))
57  throw Exception("qpp::ip()", Exception::Type::ZERO_SIZE);
58 
59  // check zero-size
60  if (!internal::_check_nonzero_size(rpsi))
61  throw Exception("qpp::ip()", Exception::Type::ZERO_SIZE);
62 
63  // check column vector
64  if (!internal::_check_cvector(rphi))
65  throw Exception("qpp::schmidtcoeffs()",
66  Exception::Type::MATRIX_NOT_CVECTOR);
67 
68  // check column vector
69  if (!internal::_check_cvector(rpsi))
70  throw Exception("qpp::schmidtcoeffs()",
71  Exception::Type::MATRIX_NOT_CVECTOR);
72 
73  // check that dims is a valid dimension vector
74  if (!internal::_check_dims(dims))
75  throw Exception("qpp::ip()", Exception::Type::DIMS_INVALID);
76 
77  // check that subsys are valid
78  if (!internal::_check_subsys_match_dims(subsys, dims))
79  throw Exception("qpp::ip()",
80  Exception::Type::SUBSYS_MISMATCH_DIMS);
81 
82  // check that dims match state vector psi
83  if (!internal::_check_dims_match_cvect(dims, rpsi))
84  throw Exception("qpp::ip()",
85  Exception::Type::DIMS_MISMATCH_CVECTOR);
86 
87  // check that subsys match state vector phi
88  std::vector<idx> subsys_dims(subsys.size());
89  for (idx i = 0; i < subsys.size(); ++i)
90  subsys_dims[i] = dims[subsys[i]];
91  if (!internal::_check_dims_match_cvect(subsys_dims, rphi))
92  throw Exception("qpp::ip()",
93  Exception::Type::DIMS_MISMATCH_CVECTOR);
94  // END EXCEPTION CHECKS
95 
96  idx Dsubsys = prod(std::begin(subsys_dims), std::end(subsys_dims));
97 
98  idx D = static_cast<idx>(rpsi.rows());
99  idx Dsubsysbar = D / Dsubsys;
100 
101  idx n = dims.size();
102  idx nsubsys = subsys.size();
103  idx nsubsysbar = n - nsubsys;
104 
105  idx Cdims[maxn];
106  idx Csubsys[maxn];
107  idx Cdimssubsys[maxn];
108  idx Csubsysbar[maxn];
109  idx Cdimssubsysbar[maxn];
110 
111  std::vector<idx> subsys_bar = complement(subsys, n);
112  std::copy(std::begin(subsys_bar), std::end(subsys_bar),
113  std::begin(Csubsysbar));
114 
115  for (idx i = 0; i < n; ++i)
116  {
117  Cdims[i] = dims[i];
118  }
119  for (idx i = 0; i < nsubsys; ++i)
120  {
121  Csubsys[i] = subsys[i];
122  Cdimssubsys[i] = dims[subsys[i]];
123  }
124  for (idx i = 0; i < nsubsysbar; ++i)
125  {
126  Cdimssubsysbar[i] = dims[subsys_bar[i]];
127  }
128 
129  auto worker = [=](idx b) noexcept
130  -> typename Derived::Scalar
131  {
132  idx Cmidxrow[maxn];
133  idx Cmidxrowsubsys[maxn];
134  idx Cmidxcolsubsysbar[maxn];
135 
136  /* get the col multi-indexes of the complement */
137  internal::_n2multiidx(b, nsubsysbar,
138  Cdimssubsysbar, Cmidxcolsubsysbar);
139  /* write it in the global row multi-index */
140  for (idx k = 0; k < nsubsysbar; ++k)
141  {
142  Cmidxrow[Csubsysbar[k]] = Cmidxcolsubsysbar[k];
143  }
144 
145  typename Derived::Scalar result = 0;
146  for (idx a = 0; a < Dsubsys; ++a)
147  {
148  /* get the row multi-indexes of the subsys */
149  internal::_n2multiidx(a, nsubsys,
150  Cdimssubsys, Cmidxrowsubsys);
151  /* write it in the global row multi-index */
152  for (idx k = 0; k < nsubsys; ++k)
153  {
154  Cmidxrow[Csubsys[k]] = Cmidxrowsubsys[k];
155  }
156  // compute the row index
157  idx i = internal::_multiidx2n(Cmidxrow, n, Cdims);
158 
159  result += std::conj(rphi(a)) * rpsi(i);
160  }
161 
162  return result;
163  }; /* end worker */
164 
165  dyn_col_vect<typename Derived::Scalar> result(Dsubsysbar);
166 #ifdef WITH_OPENMP
167 #pragma omp parallel for
168 #endif
169  for (idx m = 0; m < Dsubsysbar; ++m)
170  result(m) = worker(m);
171 
172  return result;
173 }

222 template<typename Derived>
223 std::tuple<idx, std::vector<double>, std::vector<cmat>>
224 measure(const Eigen::MatrixBase<Derived>& A, const std::vector<cmat>& Ks)
225 {
226  const dyn_mat<typename Derived::Scalar>& rA = A;
227 
228  // EXCEPTION CHECKS
229 
230  // check zero-size
231  if (!internal::_check_nonzero_size(rA))
232  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
233 
234  // check the Kraus operators
235  if (Ks.size() == 0)
236  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
237  if (!internal::_check_square_mat(Ks[0]))
238  throw Exception("qpp::measure()", Exception::Type::MATRIX_NOT_SQUARE);
239  if (Ks[0].rows() != rA.rows())
240  throw Exception("qpp::measure()",
241  Exception::Type::DIMS_MISMATCH_MATRIX);
242  for (auto&& it : Ks)
243  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
244  throw Exception("qpp::measure()", Exception::Type::DIMS_NOT_EQUAL);
245  // END EXCEPTION CHECKS
246 
247  // probabilities
248  std::vector<double> prob(Ks.size());
249  // resulting states
250  std::vector<cmat> outstates(Ks.size());
251 
252  //************ density matrix ************//
253  if (internal::_check_square_mat(rA)) // square matrix
254  {
255  for (idx i = 0; i < Ks.size(); ++i)
256  {
257  outstates[i] = cmat::Zero(rA.rows(), rA.rows());
258  cmat tmp = Ks[i] * rA * adjoint(Ks[i]); // un-normalized;
259  prob[i] = std::abs(trace(tmp)); // probability
260  if (prob[i] > eps)
261  outstates[i] = tmp / prob[i]; // normalized
262  }
263  }
264  //************ ket ************//
265  else if (internal::_check_cvector(rA)) // column vector
266  {
267  for (idx i = 0; i < Ks.size(); ++i)
268  {
269  outstates[i] = ket::Zero(rA.rows());
270  ket tmp = Ks[i] * rA; // un-normalized;
271  // probability
272  prob[i] = std::pow(norm(tmp), 2);
273  if (prob[i] > eps)
274  outstates[i] = tmp / std::sqrt(prob[i]); // normalized
275  }
276  }
277  else
278  throw Exception("qpp::measure()",
279  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
280 
281  // sample from the probability distribution
282  std::discrete_distribution<idx> dd(std::begin(prob),
283  std::end(prob));
284  idx result = dd(RandomDevices::get_instance()._rng);
285 
286  return std::make_tuple(result, prob, outstates);
287 }

367 std::tuple<idx, std::vector<double>, std::vector<cmat>>
368 measure(const Eigen::MatrixBase<Derived>& A,
369  const std::vector<cmat>& Ks,
370  const std::vector<idx>& subsys,
371  const std::vector<idx>& dims)
372 {
373  const cmat& rA = A;
374 
375  // EXCEPTION CHECKS
376 
377  // check zero-size
378  if (!internal::_check_nonzero_size(rA))
379  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
380 
381  // check that dimension is valid
382  if (!internal::_check_dims(dims))
383  throw Exception("qpp::measure()", Exception::Type::DIMS_INVALID);
384 
385  // check that dims match rho matrix
386  if (!internal::_check_dims_match_mat(dims, rA))
387  throw Exception("qpp::measure()",
388  Exception::Type::DIMS_MISMATCH_MATRIX);
389 
390  // check subsys is valid w.r.t. dims
391  if (!internal::_check_subsys_match_dims(subsys, dims))
392  throw Exception("qpp::measure()",
393  Exception::Type::SUBSYS_MISMATCH_DIMS);
394 
395  std::vector<idx> subsys_dims(subsys.size());
396  for (idx i = 0; i < subsys.size(); ++i)
397  subsys_dims[i] = dims[subsys[i]];
398 
399  idx D = prod(std::begin(dims), std::end(dims));
400  idx Dsubsys = prod(std::begin(subsys_dims), std::end(subsys_dims));
401  idx Dsubsysbar = D / Dsubsys;
402 
403  // check the Kraus operators
404  if (Ks.size() == 0)
405  throw Exception("qpp::measure()",
406  Exception::Type::ZERO_SIZE);
407  if (!internal::_check_square_mat(Ks[0]))
408  throw Exception("qpp::measure()",
409  Exception::Type::MATRIX_NOT_SQUARE);
410  if (Dsubsys != static_cast<idx>(Ks[0].rows()))
411  throw Exception("qpp::measure()",
412  Exception::Type::DIMS_MISMATCH_MATRIX);
413  for (auto&& it : Ks)
414  if (it.rows() != Ks[0].rows() || it.cols() != Ks[0].rows())
415  throw Exception("qpp::measure()",
416  Exception::Type::DIMS_NOT_EQUAL);
417  // END EXCEPTION CHECKS
418 
419  // probabilities
420  std::vector<double> prob(Ks.size());
421  // resulting states
422  std::vector<cmat> outstates(Ks.size(), cmat::Zero(Dsubsysbar, Dsubsysbar));
423 
424  //************ density matrix ************//
425  if (internal::_check_square_mat(rA)) // square matrix
426  {
427  for (idx i = 0; i < Ks.size(); ++i)
428  {
429  cmat tmp = apply(rA, Ks[i], subsys, dims);
430  tmp = ptrace(tmp, subsys, dims);
431  prob[i] = std::abs(trace(tmp)); // probability
432  if (prob[i] > eps)
433  {
434  // normalized output state
435  // corresponding to measurement result i
436  outstates[i] = tmp / prob[i];
437  }
438  }
439  }
440  //************ ket ************//
441  else if (internal::_check_cvector(rA)) // column vector
442  {
443  for (idx i = 0; i < Ks.size(); ++i)
444  {
445  ket tmp = apply(rA, Ks[i], subsys, dims);
446  prob[i] = std::pow(norm(tmp), 2);
447  if (prob[i] > eps)
448  {
449  // normalized output state
450  // corresponding to measurement result i
451  tmp /= std::sqrt(prob[i]);
452  outstates[i] = ptrace(tmp, subsys, dims);
453  }
454  }
455  }
456  else
457  throw Exception("qpp::measure()",
458  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
459 
460  // sample from the probability distribution
461  std::discrete_distribution<idx> dd(std::begin(prob),
462  std::end(prob));
463  idx result = dd(RandomDevices::get_instance()._rng);
464 
465  return std::make_tuple(result, prob, outstates);
466 }

594 measure(const Eigen::MatrixBase<Derived>& A,
595  const cmat& V,
596  const std::vector<idx>& subsys,
597  const std::vector<idx>& dims)
598 {
599  const cmat& rA = A;
600 
601  // EXCEPTION CHECKS
602 
603  // check zero-size
604  if (!internal::_check_nonzero_size(rA))
605  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
606 
607  // check that dimension is valid
608  if (!internal::_check_dims(dims))
609  throw Exception("qpp::measure()", Exception::Type::DIMS_INVALID);
610 
611  // check subsys is valid w.r.t. dims
612  if (!internal::_check_subsys_match_dims(subsys, dims))
613  throw Exception("qpp::measure()",
614  Exception::Type::SUBSYS_MISMATCH_DIMS);
615 
616  std::vector<idx> subsys_dims(subsys.size());
617  for (idx i = 0; i < subsys.size(); ++i)
618  subsys_dims[i] = dims[subsys[i]];
619 
620  idx Dsubsys = prod(std::begin(subsys_dims), std::end(subsys_dims));
621 
622  // check the matrix V
623  if (!internal::_check_nonzero_size(V))
624  throw Exception("qpp::measure()", Exception::Type::ZERO_SIZE);
625 // if (!internal::_check_square_mat(U))
626 // throw Exception("qpp::measure()", Exception::Type::MATRIX_NOT_SQUARE);
627  if (Dsubsys != static_cast<idx>(V.rows()))
628  throw Exception("qpp::measure()",
629  Exception::Type::DIMS_MISMATCH_MATRIX);
630  // END EXCEPTION CHECKS
631 
632  // number of basis (rank-1 POVM) elements
633  idx M = static_cast<idx>(V.cols());
634 
635  //************ ket ************//
636  if (internal::_check_cvector(rA))
637  {
638  const ket& rpsi = A;
639  // check that dims match state vector
640  if (!internal::_check_dims_match_cvect(dims, rA))
641  throw Exception("qpp::measure()",
642  Exception::Type::DIMS_MISMATCH_CVECTOR);
643 
644  std::vector<double> prob(M); // probabilities
645  std::vector<cmat> outstates(M); // resulting states
646 
647 #ifdef WITH_OPENMP
648 #pragma omp parallel for
649 #endif
650  for (idx i = 0; i < M; ++i)
651  outstates[i] = ip(static_cast<const ket&>(V.col(i)),
652  rpsi, subsys, dims);
653 
654  for (idx i = 0; i < M; ++i)
655  {
656  double tmp = norm(outstates[i]);
657  prob[i] = tmp * tmp;
658  if (prob[i] > eps)
659  {
660  // normalized output state
661  // corresponding to measurement result m
662  outstates[i] /= tmp;
663  }
664  }
665 
666  // sample from the probability distribution
667  std::discrete_distribution<idx> dd(std::begin(prob),
668  std::end(prob));
669  idx result = dd(RandomDevices::get_instance()._rng);
670 
671  return std::make_tuple(result, prob, outstates);
672  }
673  //************ density matrix ************//
674  else if (internal::_check_square_mat(rA))
675  {
676  // check that dims match rho matrix
677  if (!internal::_check_dims_match_mat(dims, rA))
678  throw Exception("qpp::measure()",
679  Exception::Type::DIMS_MISMATCH_MATRIX);
680 
681  std::vector<cmat> Ks(M);
682  for (idx i = 0; i < M; ++i)
683  Ks[i] = V.col(i) * adjoint(V.col(i));
684 
685  return measure(rA, Ks, subsys, dims);
686  }
687  //************ Exception: not ket nor density matrix ************//
688  throw Exception("qpp::measure()",
689  Exception::Type::MATRIX_NOT_SQUARE_OR_CVECTOR);
690 }

753 template<typename Derived>
754 std::tuple<std::vector<idx>, double, cmat>
755 measure_seq(const Eigen::MatrixBase<Derived>& A,
756  std::vector<idx> subsys,
757  std::vector<idx> dims)
758 {
759  dyn_mat<typename Derived::Scalar> cA = A;
760 
761  // EXCEPTION CHECKS
762 
763  // check zero-size
764  if (!internal::_check_nonzero_size(cA))
765  throw Exception("qpp::measure_seq()", Exception::Type::ZERO_SIZE);
766 
767  // check that dimension is valid
768  if (!internal::_check_dims(dims))
769  throw Exception("qpp::measure_seq()", Exception::Type::DIMS_INVALID);
770 
771  // check that dims match rho matrix
772  if (!internal::_check_dims_match_mat(dims, cA))
773  throw Exception("qpp::measure_seq()",
774  Exception::Type::DIMS_MISMATCH_MATRIX);
775 
776  // check subsys is valid w.r.t. dims
777  if (!internal::_check_subsys_match_dims(subsys, dims))
778  throw Exception("qpp::measure_seq()",
779  Exception::Type::SUBSYS_MISMATCH_DIMS);
780  // END EXCEPTION CHECKS
781 
782  std::vector<idx> result;
783  double prob = 1;
784 
785  // sort subsys in decreasing order,
786  // the order of measurements does not matter
787  std::sort(std::begin(subsys), std::end(subsys), std::greater<idx>{});
788 
789  //************ density matrix or column vector ************//
790  if (internal::_check_square_mat(cA) || internal::_check_cvector(cA))
