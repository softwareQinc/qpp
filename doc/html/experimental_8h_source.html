466  QCircuitDescription& gate(const cmat& U, idx i, std::string name = "") {
467  // EXCEPTION CHECKS
468 
469  try {
470  // check valid target
471  if (i > nq_)
472  throw exception::OutOfRange("qpp::QCircuitDescription::gate()");
473  // check not measured before
474  if (get_measured(i))
475  throw exception::QuditAlreadyMeasured(
476  "qpp::QCircuitDescription::gate()");
477 
478  // check square matrix for the gate
479  if (!internal::check_square_mat(U))
480  throw exception::MatrixNotSquare(
481  "qpp::QCircuitDescription::gate()");
482  // check correct dimension
483  if (static_cast<idx>(U.rows()) != d_)
484  throw exception::DimsMismatchMatrix(
485  "qpp::QCircuitDescription::gate()");
486  } catch (qpp::exception::Exception& e) {
487  std::cerr << "At STEP " << step_cnt_ << "\n";
488  throw;
489  }
490  // END EXCEPTION CHECKS
491 
492  if (name == "")
493  name = qpp::Gates::get_instance().get_name(U);
494  gates_.emplace_back(GateType::SINGLE, U, std::vector<idx>{},
495  std::vector<idx>{i}, step_cnt_++, name);
496 
497  return *this;
498  }
