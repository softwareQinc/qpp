261 std::vector<unsigned long long int> factors(unsigned long long int n)
262 {
263  if (n == 0 || n == 1)
264  throw Exception("qpp::factors()", Exception::Type::OUT_OF_RANGE);
265 
266  std::vector<unsigned long long int> result;
267  unsigned long long int d = 2;
268 
269  while (n > 1)
270  {
271  while (n % d == 0)
272  {
273  result.push_back(d);
274  n /= d;
275  }
276  ++d;
277  if (d * d > n) // changes the running time from O(n) to O(sqrt(n))
278  {
279  if (n > 1)
280  {
281  result.push_back(n);
282  }
283  break;
284  }
285  }
286 
287  return result;
288 }
