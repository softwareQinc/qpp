873  // computes the depth greedily, measuring the "height" (depth) of the
874  // "pieces" (gates) placed in a Tetris-like style
884  idx get_depth(const std::string& name = {}) const {
885  bool found = false;
886  std::vector<idx> heights(nc_ + nq_, 0);
887 
888  // iterate over all steps in the circuit
889  for (auto&& step : *this) {
890  // measurements
891  if (step.type_ == StepType::MEASUREMENT) {
892  MeasureStep measure_step = *step.measurements_ip_;
893  if (name != "" && measure_step.name_ != name)
894  continue;
895 
896  found = true; // measurement was found in the circuit
897 
898  std::vector<idx> target = measure_step.target_;
899  idx c_reg = measure_step.c_reg_;
900 
901  idx max_height = 0;
902  switch (measure_step.measurement_type_) {
903  case MeasureType::NONE:
904  case MeasureType::MEASURE_Z:
905  case MeasureType::MEASURE_Z_MANY:
906  case MeasureType::MEASURE_V:
907  case MeasureType::MEASURE_V_MANY:
908  // compute the "height" of the to-be-placed measurement
909  if (heights[c_reg] > max_height)
910  max_height = heights[c_reg];
911  for (auto&& i : target)
912  if (heights[nc_ + i] > max_height)
913  max_height = heights[nc_ + i];
914  // apply measurement
915  heights[c_reg] = max_height + 1;
916  for (auto&& i : target)
917  heights[nc_ + i] = max_height + 1;
918  break;
919  }
920  // gates
921  } else if (step.type_ == StepType::GATE) {
922  GateStep gate_step = *step.gates_ip_;
923  if (name != "" && gate_step.name_ != name)
924  continue;
925 
926  found = true; // gate was found in the circuit
927 
928  std::vector<idx> ctrl = gate_step.ctrl_;
929  std::vector<idx> target = gate_step.target_;
930  std::vector<idx> ctrl_target;
931  ctrl_target.reserve(ctrl.size() + target.size());
932  ctrl_target.insert(ctrl_target.end(), ctrl.begin(), ctrl.end());
933  ctrl_target.insert(ctrl_target.end(), target.begin(),
934  target.end());
935 
936  idx max_height = 0;
937  switch (gate_step.gate_type_) {
938  case GateType::NONE:
939  case GateType::SINGLE:
940  case GateType::TWO:
941  case GateType::THREE:
942  case GateType::CUSTOM:
943  case GateType::FAN:
944  case GateType::SINGLE_CTRL_SINGLE_TARGET:
945  case GateType::SINGLE_CTRL_MULTIPLE_TARGET:
946  case GateType::MULTIPLE_CTRL_SINGLE_TARGET:
947  case GateType::MULTIPLE_CTRL_MULTIPLE_TARGET:
948  case GateType::CUSTOM_CTRL:
949  // compute the "height" of the to-be-placed gate
950  for (auto&& i : ctrl_target)
951  if (heights[nc_ + i] > max_height)
952  max_height = heights[nc_ + i];
953  // apply (ctrl) gate
954  for (auto&& i : ctrl_target)
955  heights[nc_ + i] = max_height + 1;
956  break;
957  case GateType::SINGLE_cCTRL_SINGLE_TARGET:
958  case GateType::SINGLE_cCTRL_MULTIPLE_TARGET:
959  case GateType::MULTIPLE_cCTRL_SINGLE_TARGET:
960  case GateType::MULTIPLE_cCTRL_MULTIPLE_TARGET:
961  case GateType::CUSTOM_cCTRL:
962  // compute the "height" of the to-be-placed gate
963  for (auto&& i : ctrl)
964  if (heights[i] > max_height)
965  max_height = heights[i];
966  for (auto&& i : target)
967  if (heights[nc_ + i] > max_height)
968  max_height = heights[nc_ + i];
969  // apply classical ctrl
970  for (auto&& i : ctrl)
971  heights[i] = max_height + 1;
972  // apply gate
973  for (auto&& i : target)
974  heights[nc_ + i] = max_height + 1;
975  break;
976  } // end switch
977  } // end if (step.type_ == StepType::GATE)
978  } // end for
979 
980  return found ? *std::max_element(std::begin(heights), std::end(heights))
981  : 0;
982  }

1048  QCircuit& add_qudit(idx n = 1, idx i = -1) {
1049  // EXCEPTION CHECKS
1050 
1051  if (i == static_cast<idx>(-1)) {
1052  i = nq_;
1053  measured_.insert(std::end(measured_), n, false);
1054  } else if (i > nq_)
1055  throw exception::OutOfRange("qpp::QCircuit::add_qudit()");
1056  // END EXCEPTION CHECKS
1057 
1058  nq_ += n;
1059 
1060  // updated the measured qudits
1061  measured_.insert(std::next(std::begin(measured_), i), n, false);
1062
