205  template<typename Derived>
206  dyn_mat<typename Derived::Scalar> CTRL(const Eigen::MatrixBase <Derived>& A,
207  const std::vector <idx>& ctrl,
208  const std::vector <idx>& subsys,
209  idx n, idx d = 2) const
210  {
211  const dyn_mat<typename Derived::Scalar>& rA = A;
212 
213  // EXCEPTION CHECKS
214  // check matrix zero size
215  if (!internal::_check_nonzero_size(rA))
216  throw Exception("qpp::Gates::CTRL()", Exception::Type::ZERO_SIZE);
217 
218  // check square matrix
219  if (!internal::_check_square_mat(rA))
220  throw Exception("qpp::Gates::CTRL()",
221  Exception::Type::MATRIX_NOT_SQUARE);
222 
223  // check lists zero size
224  if (ctrl.size() == 0)
225  throw Exception("qpp::Gates::CTRL()", Exception::Type::ZERO_SIZE);
226  if (subsys.size() == 0)
227  throw Exception("qpp::Gates::CTRL()", Exception::Type::ZERO_SIZE);
228 
229  // check out of range
230  if (n == 0)
231  throw Exception("qpp::Gates::CTRL()",
232  Exception::Type::OUT_OF_RANGE);
233 
234  // check valid local dimension
235  if (d == 0)
236  throw Exception("qpp::Gates::CTRL()",
237  Exception::Type::DIMS_INVALID);
238 
239  // ctrl + gate subsystem vector
240  std::vector <idx> ctrlgate = ctrl;
241  ctrlgate.insert(std::end(ctrlgate), std::begin(subsys),
242  std::end(subsys));
243  std::sort(std::begin(ctrlgate), std::end(ctrlgate));
244 
245  std::vector <idx> dims(n, d); // local dimensions vector
246 
247  // check that ctrl + gate subsystem is valid
248  // with respect to local dimensions
249  if (!internal::_check_subsys_match_dims(ctrlgate, dims))
250  throw Exception("qpp::Gates::CTRL()",
251  Exception::Type::SUBSYS_MISMATCH_DIMS);
252 
253  // check that subsys list match the dimension of the matrix
254  if (rA.rows() != std::llround(std::pow(d, subsys.size())))
255  throw Exception("qpp::Gates::CTRL()",
256  Exception::Type::DIMS_MISMATCH_MATRIX);
257  // END EXCEPTION CHECKS
258 
259  // Use static allocation for speed!
260  idx Cdims[maxn];
261  idx midx_row[maxn];
262  idx midx_col[maxn];
263 
264  idx CdimsA[maxn];
265  idx midxA_row[maxn];
266  idx midxA_col[maxn];
267 
268  idx Cdims_bar[maxn];
269  idx Csubsys_bar[maxn];
270  idx midx_bar[maxn];
271 
272  idx ngate = subsys.size();
273  idx nctrl = ctrl.size();
274  idx nsubsys_bar = n - ctrlgate.size();
275  idx D = static_cast<idx>(std::llround(std::pow(d, n)));
276  idx DA = static_cast<idx>(rA.rows());
277  idx Dsubsys_bar = static_cast<idx>(
278  std::llround(std::pow(d, nsubsys_bar)));
279 
280  // compute the complementary subsystem of ctrlgate w.r.t. dims
281  std::vector <idx> allsubsys(n); // all subsystems
282  std::iota(std::begin(allsubsys), std::end(allsubsys), 0);
283  std::set_difference(std::begin(allsubsys), std::end(allsubsys),
284  std::begin(ctrlgate), std::end(ctrlgate),
285  std::begin(Csubsys_bar));
286 
287  for (idx k = 0; k < n; ++k)
288  {
289  midx_row[k] = midx_col[k] = 0;
290  Cdims[k] = d;
291  }
292 
293  for (idx k = 0; k < nsubsys_bar; ++k)
294  {
295  Cdims_bar[k] = d;
296  midx_bar[k] = 0;
297  }
298 
299  for (idx k = 0; k < ngate; ++k)
300  {
301  midxA_row[k] = midxA_col[k] = 0;
302  CdimsA[k] = d;
303  }
304 
305  dyn_mat<typename Derived::Scalar> result = dyn_mat<
306  typename Derived::Scalar>::Identity(D, D);
307  dyn_mat<typename Derived::Scalar> Ak;
308 
309  // run over the complement indexes
310  for (idx i = 0; i < Dsubsys_bar; ++i)
311  {
312  // get the complement's row multi-index
313  internal::_n2multiidx(i, nsubsys_bar, Cdims_bar, midx_bar);
314  for (idx k = 0; k < d; ++k)
315  {
316  Ak = powm(rA, k); // compute rA^k
317  // run over the subsys's row multi-index
318  for (idx a = 0; a < DA; ++a)
319  {
320  // get the subsys's row multi-index
321  internal::_n2multiidx(a, ngate, CdimsA, midxA_row);
322 
323  // construct the result's row multi-index
324 
325  // first the ctrl part (equal for both row and column)
326  for (idx c = 0; c < nctrl; ++c)
327  midx_row[ctrl[c]] = midx_col[ctrl[c]] = k;
328 
329  // then the complement part (equal for column)
330  for (idx c = 0; c < nsubsys_bar; ++c)
331  midx_row[Csubsys_bar[c]] = midx_col[Csubsys_bar[c]] =
332  midx_bar[c];
333 
334  // then the subsys part
335  for (idx c = 0; c < ngate; ++c)
336  midx_row[subsys[c]] = midxA_row[c];
337 
338  // run over the subsys's column multi-index
339  for (idx b = 0; b < DA; ++b)
340  {
341  // get the subsys's column multi-index
342  internal::_n2multiidx(b, ngate, CdimsA, midxA_col);
343 
344  // construct the result's column multi-index
345  for (idx c = 0; c < ngate; ++c)
346  midx_col[subsys[c]] = midxA_col[c];
347 
348  // finally write the values
349  result(internal::_multiidx2n(midx_row, n, Cdims),
350  internal::_multiidx2n(midx_col, n, Cdims))
351  = Ak(a, b);
352  }
353  }
354 
355  }
356  }
357 
358  return result;
359  }

375  template<typename Derived>
376  dyn_mat<typename Derived::Scalar> expandout(
377  const Eigen::MatrixBase <Derived>& A, idx pos,
378  const std::vector <idx>& dims) const
379  {
380  const dyn_mat<typename Derived::Scalar>& rA = A;
381 
382  // check zero-size
383  if (!internal::_check_nonzero_size(rA))
384  throw Exception("qpp::Gates::expandout()",
385  Exception::Type::ZERO_SIZE);
386 
387  // check that dims is a valid dimension vector
388  if (!internal::_check_dims(dims))
389  throw Exception("qpp::Gates::expandout()",
390  Exception::Type::DIMS_INVALID);
391 
392  // check square matrix
393  if (!internal::_check_square_mat(rA))
394  throw Exception("qpp::Gates::expandout()",
395  Exception::Type::MATRIX_NOT_SQUARE);
396 
397  // check that position is valid
398  if (pos > dims.size() - 1)
399  throw Exception("qpp::Gates::expandout()",
400  Exception::Type::OUT_OF_RANGE);
401 
402  // check that dims[pos] match the dimension of A
403  if (static_cast<idx>(rA.rows()) != dims[pos])
404  throw Exception("qpp::Gates::expandout()",
405  Exception::Type::DIMS_MISMATCH_MATRIX);
406 
407  auto multiply = [](idx x, idx y) -> idx
408  {
409  return x * y;
410  };
411 
412  idx D = std::accumulate(std::begin(dims), std::end(dims),
413  1u, multiply);
414  dyn_mat<typename Derived::Scalar> result = dyn_mat<
415  typename Derived::Scalar>::Identity(D, D);
416 
417  idx Cdims[maxn];
418  idx midx_row[maxn];
419  idx midx_col[maxn];
420 
421  for (idx k = 0; k < dims.size(); ++k)
422  {
423  midx_row[k] = midx_col[k] = 0;
424  Cdims[k] = dims[k];
425  }
426 
427  // run over the main diagonal multi-indexes
428  for (idx i = 0; i < D; ++i)
429  {
430  // get row multi_index
431  internal::_n2multiidx(i, dims.size(), Cdims, midx_row);
432  // get column multi_index (same as row)
433  internal::_n2multiidx(i, dims.size(), Cdims, midx_col);
434  // run over the gate's row multi-index
435  for (idx a = 0; a < static_cast<idx>(rA.rows());
436  ++a)
437  {
438  // construct the total row multi-index
439  midx_row[pos] = a;
440 
441  // run over the gate's column multi-index
442  for (idx b = 0;
443  b < static_cast<idx>(rA.cols()); ++b)
444  {
445  // construct the total column multi-index
446  midx_col[pos] = b;
447 
448  // finally write the values
449  result(internal::_multiidx2n(midx_row, dims.size(), Cdims),
450  internal::_multiidx2n(midx_col, dims.size(), Cdims))
451  = rA(a, b);
452  }
453  }
454  }
455 
456  return result;
457  }
